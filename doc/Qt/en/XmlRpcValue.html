class XmlRpcValue
{
  public:

    enum Type             {
      TypeInvalid  =  0   ,
      TypeBoolean  =  1   ,
      TypeInt      =  2   ,
      TypeDouble   =  3   ,
      TypeString   =  4   ,
      TypeDateTime =  5   ,
      TypeBase64   =  6   ,
      TypeArray    =  7   ,
      TypeStruct   =  8   ,
      TypeTuid     =  9   ,
      TypeUuid     = 10   ,
      TypeEnd      = 11 } ;

    typedef QByteArray                    BinaryData  ;
    typedef QList < XmlRpcValue         > ValueArray  ;
    typedef QMap  < QString,XmlRpcValue > ValueStruct ;

  protected:

    Type _type               ;
    union                    {
      bool          asBool   ;
      int           asInt    ;
      qint64        asTuid   ;
      quint64       asUuid   ;
      double        asDouble ;
      QDateTime   * asTime   ;
      QString     * asString ;
      BinaryData  * asBinary ;
      ValueArray  * asArray  ;
      ValueStruct * asStruct ;
    } _value                 ;

  public:

    explicit XmlRpcValue (void                                  ) ;
    explicit XmlRpcValue (bool                value             ) ;
    explicit XmlRpcValue (int                 value             ) ;
    explicit XmlRpcValue (qint64              value             ) ;
    explicit XmlRpcValue (quint64             value             ) ;
    explicit XmlRpcValue (double              value             ) ;
    explicit XmlRpcValue (QDateTime           dateTime          ) ;
    explicit XmlRpcValue (const QString     & value             ) ;
    explicit XmlRpcValue (const char        * value             ) ;
    explicit XmlRpcValue (void              * value,int   nBytes) ;
    explicit XmlRpcValue (const QString     & xml  ,int * offset) ;
             XmlRpcValue (const XmlRpcValue & rhs               ) ;
    virtual ~XmlRpcValue (void                                  ) ;

    virtual void  clear      (void) ;

    XmlRpcValue & operator = (const XmlRpcValue & rhs) ;
    XmlRpcValue & operator = (const bool        & rhs) ;
    XmlRpcValue & operator = (const int         & rhs) ;
    XmlRpcValue & operator = (const qint64      & rhs) ;
    XmlRpcValue & operator = (const quint64     & rhs) ;
    XmlRpcValue & operator = (const double      & rhs) ;
    XmlRpcValue & operator = (const char        * rhs) ;
    XmlRpcValue & operator = (const QString     & rhs) ;
    XmlRpcValue & operator = (const QDateTime   & rhs) ;

    bool operator == (const XmlRpcValue & other) const ;
    bool operator != (const XmlRpcValue & other) const ;

    const Type & getType (void) const ;

    operator bool        & ( ) { assertTypeOrInvalid(TypeBoolean ) ; return  _value.asBool   ; }
    operator int         & ( ) { assertTypeOrInvalid(TypeInt     ) ; return  _value.asInt    ; }
    operator qint64      & ( ) { assertTypeOrInvalid(TypeTuid    ) ; return  _value.asTuid   ; }
    operator quint64     & ( ) { assertTypeOrInvalid(TypeUuid    ) ; return  _value.asUuid   ; }
    operator double      & ( ) { assertTypeOrInvalid(TypeDouble  ) ; return  _value.asDouble ; }
    operator QString     & ( ) { assertTypeOrInvalid(TypeString  ) ; return *_value.asString ; }
    operator BinaryData  & ( ) { assertTypeOrInvalid(TypeBase64  ) ; return *_value.asBinary ; }
    operator QDateTime   & ( ) { assertTypeOrInvalid(TypeDateTime) ; return *_value.asTime   ; }

    const XmlRpcValue    & operator [] (int i) const ;
          XmlRpcValue    & operator [] (int i) ;

    XmlRpcValue & operator [] (const QString & k) ;
    XmlRpcValue & operator [] (const char    * k) ;

    bool           valid     (void) const ;
    int            size      (void) const ;
    void           setSize   (int size) ;

    bool           hasMember (const QString & name) const ;

    bool           fromXml   (const QString & valueXml,int * offset) ;
    QString        toXml     (void) const ;

    std::ostream & write     (std::ostream& os) const ;

    static QString const & getDoubleFormat (void) ;
    static void            setDoubleFormat (const char * f) ;

  protected:

    void    invalidate          (void) ;

    void    assertTypeOrInvalid (Type t) ;
    void    assertArray         (int size) ;
    void    assertStruct        (void) ;

    bool    boolFromXml         (const QString & valueXml,int * offset) ;
    bool    intFromXml          (const QString & valueXml,int * offset) ;
    bool    tuidFromXml         (const QString & valueXml,int * offset) ;
    bool    uuidFromXml         (const QString & valueXml,int * offset) ;
    bool    doubleFromXml       (const QString & valueXml,int * offset) ;
    bool    stringFromXml       (const QString & valueXml,int * offset) ;
    bool    timeFromXml         (const QString & valueXml,int * offset) ;
    bool    binaryFromXml       (const QString & valueXml,int * offset) ;
    bool    arrayFromXml        (const QString & valueXml,int * offset) ;
    bool    structFromXml       (const QString & valueXml,int * offset) ;

    QString boolToXml           (void) const ;
    QString intToXml            (void) const ;
    QString tuidToXml           (void) const ;
    QString uuidToXml           (void) const ;
    QString doubleToXml         (void) const ;
    QString stringToXml         (void) const ;
    QString timeToXml           (void) const ;
    QString binaryToXml         (void) const ;
    QString arrayToXml          (void) const ;
    QString structToXml         (void) const ;

    static QString _doubleFormat ;

  private:

} ;
