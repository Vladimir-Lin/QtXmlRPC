{

    int              getfd          (void) const ;

    bool             getStopped     (void) const ;
    void             setStopped     (bool b = true) ;
    virtual void     close          (void) ;
    virtual unsigned handleEvent    (unsigned eventType) = 0 ;

  protected:

    int    READSIZE ;
    char * readBuf  ;

  private:

    int  fd            ;
    bool deleteOnClose ;
    bool keepOpen      ;
    bool Stopped       ;

} ;
